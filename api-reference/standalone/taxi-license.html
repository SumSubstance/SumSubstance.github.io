# Supplemental endpoints

# Taxi license information

POST /resources/checks/taxiLicenseInfo

# REQUEST ARGUMENTS
Name Type Required Description
country String Yes Country
additionalNumber String Yes Vehicle registration number
# RESPONSE
Name Type Required Description
licenseValidity Boolean Yes Binary answer about license validity
licenseNumber String No License number
name String No The name of the organization or individual
licensePlateNumber String No Vehicle registration number
brand String No Vehicle manufacturer
# Example request
Copy
curl -X POST \
  'https://api.sumsub.com/resources/checks/taxiLicenseInfo' \
  -H 'Content-Type: application/json' \
  -d '{
          "country": "RUS",
          "additionalNumber": "В731АР716"
      }'
# Example response
Copy
{
    "licenseValidity": true,
    "licenseNumber": "0038063",
    "name": "Кольт Диана Фердинантовна",
    "licensePlateNumber": "В731АР716",
    "brand": "KIA RIO"
}
