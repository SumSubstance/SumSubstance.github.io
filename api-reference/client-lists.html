# Client lists

Client lists are used for KYT and Workflow rules/condition. You can manage them through the dashboard.

# Client lists API reference

All API requests must be authenticated as described in this section.

# Getting all client lists

Retrieves the full list of existing client lists.

GET /resources/clientLists/-

# RESPONSE

Response represents a list of client list data (.list.items).

# .items values attributes
Name Type Description
id String Client list ID.
clientId String Unique identifier of you as our client.
createdAt Date Time and date of the client list creation.
name String Client list name.
type String Client list type. May be one of three values: blocklist/whitelist/custom.
entryType String Entry type. Describes the type of values that list contains: key/keyValue/applicant,email/deviceFingerprint.
title String Client list title.
desc String Client list description.
# Example request
Copy
curl -X GET \
  'https://api.sumsub.com/resources/clientLists/-'
# Example response
Copy
{
  "list": {
    "items": [ {
      "id": "64c8d21e8dab4e6826fe229b",
      "createdAt": "2023-08-01 09:36:30+0000",
      "clientId": "coolClientId",
      "name": "my_blocklist_x",
      "type": "blocklist",
      "entryType": "applicant",
      "title": "My Blocklist X",
      "desc": "Blocklist description"
    }, {
      "id": "64c76ec9b77e490ff442227d",
      "createdAt": "2023-07-31 08:20:25+0000",
      "clientId": "coolClientId",
      "name": "high_risk_countries",
      "type": "custom",
      "entryType": "key",
      "title": "High Risk Countries",
      "desc": "High Risk Countries"
    } ],
    "totalItems": 2
  }
}

# Getting client list entries

GET /resources/clientLists/{clientListId}/values?offset={offset}&limit={limit}

# REQUEST ARGUMENTS
Name Type Required Description Default
clientListId String Yes Client list ID.
offset Integer No Offset of client list entries to be returned. 0
limit Integer No Max number of client list entries in one request. 10
# RESPONSE

Response represents a list of client list entry data (.list.items).

# .items values attributes
Name Type Description
id String Client list entry ID.
clientId String Unique identifier of you as our client.
createdAt Date Time and date of the client list entry creation.
createdBy String Entry creator's account name.
listName String Client list name.
key String Entity key.
slug String Attribute that is used for better understanding what's inside, e.g. full applicant name for applicant list.
desc String Client list entry description.
# Example request
Copy
curl -X GET \
  'https://api.sumsub.com/resources/clientLists/64c76ec9b77e490ff442227d/values?offset=0&limit=100'
# Example response
Copy
{
  "list": {
    "items": [ {
      "id": "64c8a04726d81f4d4dc91fbe",
      "createdAt": "2023-08-01 06:03:52+0000",
      "createdBy": "Service",
      "clientId": "coolClientId",
      "listName": "high_risk_countries",
      "key": "AGO",
      "desc": "Angola"
    }, {
      "id": "64c8a04726d81f4d4dc91fbf",
      "createdAt": "2023-08-01 06:03:51+0000",
      "createdBy": "Service",

# Adding client list entry

POST /resources/clientLists/{clientListId}/values

If the provided key already exists, it will be replaced with a new value.

# REQUEST ARGUMENTS
Name Type Required Description
clientListId String Yes Client list ID.
#{body} Object Yes Entry data.
# Entry data attributes
Name Type Description
id String Client list entry ID.
clientId String Unique identifier of you as our client.
createdAt Date Time and date of the client list entry creation.
createdBy String Entry creator's account name.
listName String Client list name.
key String Entity key .
slug String Attribute that is used for better understanding what's inside, e.g. full applicant name for applicant list.
desc String Client list entry description.
# RESPONSE

Response represents a client list entry data.

# Example request
Copy
curl -X POST \
  'https://api.sumsub.com/resources/clientLists/64c76ec9b77e490ff442227d/values' \
  -H 'Content-Type: application/json' \
  -d '{
        "key": "ATA",
        "desc": "Antarctica"
      }'
# Example response
Copy
{
  "id": "64c8d56e8dab4e6826fe22a2",
  "createdAt": "2023-08-01 09:50:38+0000",
  "createdBy": "Service",
  "clientId": "coolClientId",
  "listName": "high_risk_countries",
  "key": "ATA",
  "desc": "Antarctica"
}

# Removing client list entry

DELETE /resources/clientLists/{clientListId}/values?key={key}

# REQUEST ARGUMENTS
Name Type Required Description
clientListId String Yes Client list ID.
key String Yes Entry key that should be removed.
# Example request
Copy
curl -X DELETE \
  'https://api.sumsub.com/resources/clientLists/64c76ec9b77e490ff442227d/values?key=ATA'
# Example response
Copy
{ "ok": 1 }
